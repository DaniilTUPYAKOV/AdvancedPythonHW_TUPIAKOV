    65  from collections.abc import Iterable
    66
    67  def process_lines(start_index: int, source: Iterable[str]):
    68      """Processing lines in iterable line source
    69
    70      Args:
    71          start_index (int): start of numeration
    72          source (Iterable[str]): iterable source of lines (str type)
    73      """
    74      index = start_index
    75      for line in source:
    76          formatted_print(index, line)
    77          index+=1
    78
    79  def formatted_print(number: int, text: str):
    80      """ Print given text with given number prefix in specified format
    81
    82      Args:
    83          number (int): number to be printed as prefix for text
    84          text (str): tetx to be printed
    85      """
    86      space_for_digits_in_number = 6
    87      number_length = len(str(number))
    88      prefix = ""
    89
    90      if number_length <= space_for_digits_in_number:
    91          prefix = "".join([" "]*(space_for_digits_in_number-number_length))
    92
    93      print(f'{prefix}{number}  {text}', end="")
    94
    95  def simple_nl_in_python():

   100      if len(sys.argv) == 1:

   102      else:
   103          for filename in sys.argv[1:]:
   104              with open(filename, 'r',encoding="UTF-8") as file:
   105                  source+=file.readlines()
   106                  file.close()
   107
   108      process_lines(start_index, source)
